// 在以root为根的树中找到值为val1或val2的节点

const find = (root: TreeNode | null, val1: number, val2: number): TreeNode | null => {
  // base case
  if (root === null)
    return null

  // 前序位置看root是不是目标节点
  if (root.val === val1 || root.val === val2)
    return root

  // 去左右子树找
  const left = find(root.left, val1, val2)
  const right = find(root.right, val1, val2)

  // 后续位置，已经知道左右子树是否存在目标值
  return left !== null ? left : right 
}
