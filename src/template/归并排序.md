function MergeSort(nums: number[]) {
  const temp = new Array<number>()
  sort(nums, 0, nums.length - 1, temp)
}

function sort(nums: number[], lo: number, hi: number, temp: number[]) {
  if (lo == hi) {
    return
  }

  const mid = lo + Math.floor((hi - lo) / 2)

  sort(nums, lo, mid, temp)
  sort(nums, mid + 1, hi, temp)

  merge(nums, lo, mid, hi, temp)
}

function merge(nums: number[], lo: number, mid: number, hi: number, temp: number[]) {
  for (let i = lo; i <= hi; i++) {
    temp[i] = nums[i]
  }

  let i = lo, j = mid + 1
  for (let p = lo; p <= hi; p++) {
    if (i === mid + 1) {
      nums[p] = temp[j]
      j++
    } else if (j === hi + 1) {
      nums[p] = temp[i]
      i++
    } else if (temp[i] <= temp[j]) {
      nums[p] = temp[i]
      i++
    } else {
      nums[p] = temp[j]
      j++
    }
  }
}
