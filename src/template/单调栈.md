// 单调栈只处理一类典型的问题，比如「下一个更大元素」，「上一个更小元素」等。

// 例：输入一个数组 nums，请你返回一个等长的结果数组，结果数组中对应索引存储着下一个更大元素，如果没有更大的元素，就存 -1。

const calculateGreaterElement = function (nums: number[]): number[] {
  const n = nums.length
  // 存放答案的数组
  const res: number[] = new Array(n)
  const stk: number[] = []
  // 倒着往栈里放
  for (let i = n - 1; i >= 0; i--) {
    // 判断个子高矮
    while (stk.length !== 0 && stk[stk.length - 1] <= nums[i]) {
      // 矮个子起开
      stk.pop()
    }
    // nums[i] 身后更大的元素
    res[i] = stk.length === 0 ? -1 : stk[stk.length - 1]
    stk.push(nums[i])
  }

  return res
}
