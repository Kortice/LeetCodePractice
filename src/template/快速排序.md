function QuickSort(nums: number[]) {
  shuffle(nums)
  sort(nums, 0, nums.length - 1)
}

function sort(nums: number[], lo: number, hi: number) {
  if (lo >= hi) {
    return
  }
  const p = partition(nums, lo, hi)
  sort(nums, lo, p - 1)
  sort(nums, p + 1, hi)
}

function partition(nums: number[], lo: number, hi: number): number {
  const pivot = nums[lo]
  let i = lo + 1, j = hi
  while (i <= j) {
    while (i < hi && nums[i] <= pivot) {
      i++
    }
    while (j > lo && nums[j] > pivot) {
      j--
    }
    if (i >= j) {
      break
    }
    swap(nums, i, j)
  }
  swap(nums, lo, j)
  return j
}

function shuffle(nums: number[]) {
  const n = nums.length
  for (let i = 0; i < n; i++) {
    const r = Math.floor(Math.random() * (n - i)) + i
    swap(nums, i, r)
  }
}

function swap(nums: number[], i: number, j: number) {
  [nums[i], nums[j]] = [nums[j], nums[i]]
}
