// 建议优先队列实现

class SimpleMinPQ {
  // 底层数组
  heap: number[]
  // 堆的大小
  _size: number

  constructor(capacity: number) {
    this.heap = new Array(capacity)
    this._size = 0
  }

  // 返回当前节点的父节点
  parent(node: number): number {
    return Math.floor((node - 1) / 2)
  }

  // 返回当前节点的左子节点
  left(node: number): number {
    return node * 2 + 1
  }

  // 返回当前节点的右子节点
  right(node: number): number {
    return node * 2 + 2
  }

  // 交换数据
  swap(n1: number, n2: number) {
    [this.heap[n1], this.heap[n2]] = [this.heap[n2], this.heap[n1]]
  }

  // 推element入堆
  push(element: number) {
    this.heap[this._size] = element
    // 上浮操作
    this.swim(this._size)
    this._size++
  }

  // 推出堆顶元素
  pop(): number {
    const res = this.heap[0]
    this.swap(this._size - 1, 0)
    this._size--
    // 下沉操作
    this.sink(0)

    return res
  }

  // 返回堆顶元素
  peek(): number {
    return this.heap[0]
  }

  // 返回堆大小
  size(): number {
    return this._size
  }

  // 上浮操作
  swim(node: number) {
    // 还没到根节点 & 父节点大于自己
    while (node > 0 && this.heap[this.parent(node)] > this.heap[node]) {
      // 交换节点
      this.swap(node, this.parent(node))
      // 更新指针
      node = this.parent(node)
    }
  }

  // 下沉操作
  sink(node: number) {
    // 存在子节点
    while (this.left(node) < this._size || this.right(node) < this._size) {
      // 找最小
      let min = node
      if (this.left(node) < this._size && this.heap[this.left(node)] < this.heap[min]) {
        min = this.left(node)
      }
      if (this.right(node) < this._size && this.heap[this.right(node)] < this.heap[min]) {
        min = this.right(node)
      }

      // 当前已经是最小了
      if (min === node) {
        return
      }

      // 交换节点
      this.swap(node, min)

      // 更新指针
      node = min
    }
  }
}
