// 前缀和技巧适用于快速、频繁地计算一个索引区间内的元素之和。

// 前缀和数组 preSum[i] 表示的是 前面i个元素 的和

// 返回一个数组的前缀和数组
function NumArray(nums: number[]): number[] {

  // 前缀和数组
  let preSum = new Array(nums.length + 1).fill(0)
  // 便于计算累加和
  preSum[0] = 0

  // 输入一个数组，构造前缀和
  for (let i = 1; i < preSum.length; i++) {
    // 计算 nums 的累加和
    preSum[i] = preSum[i - 1] + nums[i - 1]
  }

  return preSum
} // preSum[i] = nums[0] + nums[1] + ... + nums[i-1]

// 求区间的和 [left, right]
function sumRange(preSum: number[], left: number, right: number): number {
  return preSum[right + 1] - preSum[left]
}
